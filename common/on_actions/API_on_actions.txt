# Root = Character
api_on_character_death = {
    effect = {
        if = {
            limit = {
                has_variable = api_character_bolstered
            }
            owner = {
                set_local_variable = {
                    name  = qty_delta
                    value = -1
                }
                trigger_event = {
                    id = api_ev.2
                }
            }
        }
        if = {
            limit = {
                has_variable = api_character_suppressed
            }
            owner = {
                set_local_variable = {
                    name  = qty_delta
                    value = -1
                }
                trigger_event = {
                    id = api_ev.3
                }
            }
        }
    }
}
on_character_death = {
    on_actions = {
        api_on_character_death
    }
}
