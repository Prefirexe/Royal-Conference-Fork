je_american_isolationism = {
	icon = "gfx/interface/icons/event_icons/usa_isolation.dds"
	
	group = je_group_historical_content
	
	immediate = {
		set_variable = {
			name = usa_infamy
			value = 0
		}
		add_modifier = {
			name = usa_isolationism
		}
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	
	on_complete = {
		hidden_effect = {
			remove_modifier = usa_notoriety
		}
		remove_modifier = usa_isolationism
		add_modifier = {
			name = usa_imperialism		
			is_decaying = yes
			years = 25
		}
		add_modifier = {
			name = usa_notoriety			
			is_decaying = yes
			years = 10
		}		
	}
	
	fail = {
		OR = {
			NOT = { has_law = law_type:law_republic }
			AND = {
				NOT = { has_law = law_type:law_right_of_assembly }
				NOT = { has_law = law_type:law_protected_speech }
				NOT = { has_law = law_type:law_census_voting }
				NOT = { has_law = law_type:law_universal_suffrage }
			}
		}
	}
	
	on_fail = {
		remove_modifier = usa_isolationism
		remove_modifier = usa_notoriety
	}

	on_monthly_pulse = {
		effect = {
			set_variable = {
				name = usa_infamy
				value = infamy
			}
			remove_modifier = usa_notoriety
			if = {
				limit = {
					infamy >= 10
					infamy < 20
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.1
				}
			}
			if = {
				limit = {
					infamy >= 20
					infamy < 30
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.2
				}
			}
			if = {
				limit = {
					infamy >= 30
					infamy < 40
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.3
				}
			}
			if = {
				limit = {
					infamy >= 40
					infamy < 50
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.4
				}
			}
			if = {
				limit = {
					infamy >= 50
					infamy < 60
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.5
				}
			}
			if = {
				limit = {
					infamy >= 60
					infamy < 70
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.6
				}
			}
			if = {
				limit = {
					infamy >= 70
					infamy < 80
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.7
				}
			}
			if = {
				limit = {
					infamy >= 80
					infamy < 90
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.8
				}
			}
			if = {
				limit = {
					infamy >= 90
				}
				add_modifier = {
					name = usa_notoriety
					multiplier = 0.9
				}
			}
		}
	}

	current_value = {
		value = root.var:usa_infamy
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	can_revolution_inherit = yes

	weight = 5000
	should_be_pinned_by_default = yes
}
