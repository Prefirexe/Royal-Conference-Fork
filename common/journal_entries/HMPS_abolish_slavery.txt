# triggered by on_law_enactment_started
je_abolish_slavery = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	group = je_group_internal_affairs

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_law = law_type:law_slavery_banned
					NOT = {
						any_civil_war = {
							civil_war_progress >= 0.75
							is_civil_war_type = revolution
						}
					}
					NOT = {
						any_interest_group = {
							OR = {
								law_stance = {
									law = law_type:law_slave_trade
									value > neutral
								}
								law_stance = {
									law = law_type:law_debt_slavery
									value > neutral
								}
								law_stance = {
									law = law_type:law_legacy_slavery
									value > neutral
								}
							}
							is_powerful = yes
						}
					}
				}
				change_variable = {
					name = abolishing_slavery_var
					add = 1
				}
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		has_law = law_type:law_slavery_banned
		NOT = {
			any_civil_war = {
				civil_war_progress >= 0.75
				is_civil_war_type = revolution
			}
		}
		NOT = {
			any_interest_group = {
				OR = {
					law_stance = {
						law = law_type:law_slave_trade
						value > neutral
					}
					law_stance = {
						law = law_type:law_debt_slavery
						value > neutral
					}
					law_stance = {
						law = law_type:law_legacy_slavery
						value > neutral
					}
				}
				is_powerful = yes
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = 1848.7
		}
	}

	invalid = {
		NOR = {
			is_enacting_law = law_type:law_slavery_banned
			has_law = law_type:law_slavery_banned
		}
	}

	current_value = {
		value = root.var:abolishing_slavery_var
	}

	goal_add_value = {
		add = 60
	}

	progressbar = yes

	weight = 100
	
	should_be_pinned_by_default = yes
}
